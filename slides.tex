\subsection{Example}

\begin{frame}[fragile]{Regions: Translation Example}
\begin{lstlisting}
let X = (2, 3) in (\:Y.(fst X, Y)) end 5
\end{lstlisting}
\pause
Gets translated to:

\begin{lstlisting}
letregion R4, R5
in letregion R6
   in let X = (2 at R2, 3 at R6) at R4
      in (\:Y.(fst X, Y) at R1) at R5
      end
   end
   5 at R3
end
\end{lstlisting}
\end{frame}

\begin{frame}[fragile]{Regions: Translation Example}
\begin{lstlisting}
(@\snippet{let $x$ = (2, 3) in ($\lambda y$.(fst $x$, $y$)) end 5}@)
\end{lstlisting}

Gets translated to:

\begin{lstlisting}
letregion R4, R5
in letregion R6
   in (@\snippet{let $x$ = (2}@) at R2(@\snippet{, 3}@) at R6(@\snippet{)}@) at R4
      (@\snippet{in ($\lambda y$.(fst $x$, $y$)}@) at R1(@\snippet{)}@) at R5
      (@\snippet{end}@)
   end
   (@\snippet{5}@) at R3
end
\end{lstlisting}
\end{frame}

        \pause
